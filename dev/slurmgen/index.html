Let's generate the commands we need for likelihood spectra.

mapids = [k for k in keys(config["map"])]
run = "sbatch scripts/4core1hr.cmd"
for i in 1:length(mapids)
    for j in i:length(mapids)
        println("$(run) \"julia src/rawspectra.jl global.toml $(mapids[i]) $(mapids[j])\"")
    end
end

Fit Noise Model

These SLURM commands fit the raw spectra with the camspec noise model.

# loop over freqs and noise channels
for freq in ("100", "143", "217")
    for spec in ("TT", "EE")
        println("$(run) \"julia src/fitnoisemodel.jl global.toml $(freq) $(spec)\"")
    end
end
sbatch scripts/4core1hr.cmd "julia src/fitnoisemodel.jl global.toml 100 TT"
sbatch scripts/4core1hr.cmd "julia src/fitnoisemodel.jl global.toml 100 EE"
sbatch scripts/4core1hr.cmd "julia src/fitnoisemodel.jl global.toml 143 TT"
sbatch scripts/4core1hr.cmd "julia src/fitnoisemodel.jl global.toml 143 EE"
sbatch scripts/4core1hr.cmd "julia src/fitnoisemodel.jl global.toml 217 TT"
sbatch scripts/4core1hr.cmd "julia src/fitnoisemodel.jl global.toml 217 EE"
